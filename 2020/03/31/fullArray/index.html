有重复元素的全排列



//(Description：//)有n个字符，其中可能有重复字符，求出它们的全排列并输出，不输出重复排列，并最后输出总共不重复序列的个数。

一个排列按照字典序的下一个排列：从最右边开始寻找，找到第一个相邻两个数从左到右是递增的，令左边元素为a，右边元素为b，在a的右边找到比a大的最小的数和a进行交换，如果多个数并列最小，选择靠右端的数和a交换，然后将刚刚a位置右边的所有的数字反转。

写递归代码的关键就是找到如何将大问题分解为小问题，并且基于此写出递归公式，然后再推敲终止条件，不用去想一层层的调用关系，不要试图用人脑去分解递归的每个步骤。

#include< iostream>
using namespace std;

char* swap(char* a, int b, int c);
void fullArray(char* a, int s, int e);
bool isRepeat(char* a, int x, int y);
int sum = 0;
int main() {
    int n;
    cin >> n;
    char* a = new char[n];
    cin >> a;
    fullArray(a, 0, n - 1);
    cout << sum;
    return 0;
}

void fullArray(char* a, int s, int e) {
    if (s == e) {
        cout << a << endl;
        sum++;
    }
    else {
        for (int i = s; i <= e; i++) {
            if (!isRepeat(a, s, i)) { //如果s[i]前面的数和s[i]相等则不交换
                a = swap(a, s, i);  //从原本的序列开始
                fullArray(a, s + 1, e);
                a = swap(a, i, s);  //保证进入下一次循环时是原来的序列
            }
        }
    }
}

char* swap(char* a, int b, int c) {
    char t;
    t = a[b];
    a[b] = a[c];
    a[c] = t;
    return a;
}

bool isRepeat(char* a, int x, int y) {
    for (int i = x; i < y; i++) {
        if (a[i] == a[y]) {
            return true;
        }
    }
    return false;
}

Postscript：将左端的数和右边的数依次交换即得到左端位的所有可能，然后进入相邻右端的下一位，进行递归过程，在与右边的数进行交换的时候，如果右边的数字的左边有和它相同的数字，则不进入循环，因为交换的结果之前已经出现，再次交换会产生重复的序列。

void fullArray(char* t, int n) {
    sort(t, t + n);
    cout << t << endl;
    int e = n - 1;
    int x, y, k, w;
    int m;
    while (1) {
        y = e - 1;
        while (t[y] >= t[y + 1]) {
            y--;
            if (y < 0) {
                break;
            }
        }
        if (y < 0) {
            break;
        }
        x = y;
        for (k = e; k > x; k--) {
            if (t[k] > t[x]) {
                m = k;
                break;
            }
        }
        for (k = m - 1; k > x; k--) {
            if (t[k] < t[m]) {
                m = k;
            }
        }
        swap(t[x], t[m]);
        reverse(t + x + 1, t + n);
        cout << t << endl;
        sum++;
    }
}
