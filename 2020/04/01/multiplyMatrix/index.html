矩阵相乘



\(Description:\)分别输入两个矩阵，输出两个矩阵相乘后得到的矩阵。

#include<iostream>
using namespace std;

void inputMatrix(int**, int, int);
int** multiplyMatrix(int**, int**, int, int, int);
void outputMatrix(int**, int, int);
int main() {
	int m, n, p; //m为左边矩阵的行数，n为左边矩阵的列数和右边矩阵的行数，p为右边矩阵的列数
	cin >> m >> n >> p;
	int** matrix1 = new int* [m];
	int** matrix2 = new int* [n];
	for (int i = 0; i < m; i++) {
		*(matrix1 + i) = new int[n];
	}
	for (int j = 0; j < n; j++) {
		*(matrix2 + j) = new int[p];
	}
	inputMatrix(matrix1, m, n);
	inputMatrix(matrix2, n, p);
	int** matrix3;
	matrix3 = multiplyMatrix(matrix1, matrix2, m, n, p);
	outputMatrix(matrix3, m, p);
	for (int i = 0; i < m; i++) {
		delete[] * (matrix1 + i);
	}
	delete[] matrix1;
	for (int j = 0; j < n; j++) {
		delete[] * (matrix2 + j);
	}
	delete[] matrix2;
	return 0;
}

void inputMatrix(int** matrix, int x, int y) {
	for (int i = 0; i < x; i++) {
		for (int j = 0; j < y; j++) {
			cin >> matrix[i][j];
		}
	}
}

int** multiplyMatrix(int** matrix1, int** matrix2, int m, int n, int p) {
	int** matrix3 = new int* [m];
	int ans = 0;
	for (int i = 0; i < m; i++) {
		*(matrix3 + i) = new int[p];
	}
	for (int x = 0; x < m; x++) {
		for (int y = 0; y < p; y++) {
			ans = 0;
			for (int k = 0; k < n; k++) {
				ans += matrix1[x][k] * matrix2[k][y];
			}
			matrix3[x][y] = ans;
		}
	}
	return matrix3;
}

void outputMatrix(int** matrix, int x, int y) {
	int i, j;
	for (i = 0; i < x; i++) {
		for (j = 0; j < y - 1; j++) {
			cout << matrix[i][j] << " ";
		}
		cout << matrix[i][j] << endl;
	}
}
