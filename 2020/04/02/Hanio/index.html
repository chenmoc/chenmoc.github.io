汉诺塔



\(Description:\)输入n为汉诺塔盘子的数目，输出盘子移动的步骤和步骤的总数。

#include<iostream>
using namespace std;

long long sum = 0;
void hanio(int n,char a,char b,char c) {
	if (n == 1) {  //当只有一个盘子时为递归的终止条件
		cout << a << "->" << c << endl;
		sum++;
	}
	else {
		hanio(n - 1, a, c, b);  //将上方n-1个盘子从A移动到B上
		cout << a << "->" << c << endl;  //将底端的盘子移动到C上
		sum++;
		hanio(n - 1, b, a, c);  //将上方n-1个盘子从B移动到C上
	}
}

int main(void) {
	int n;
	char a = 'A', b = 'B', c = 'C';
	cin >> n;
	hanio(n, a, b, c);
	cout << sum;
}

\(Postcript:\)汉诺塔挪动n个盘子的步骤总数为\(2^n{-}1\)
